class Solution {
public:
    int n,m;
    vector<int> findBall(vector<vector<int>>& grid) {
        vector<int> ret;
        n = grid[0].size();
        m = grid.size();
        for(int i = 0;i<n;i++){
            ret.push_back(judge(0,i,grid));
        }
        return ret;
    }
    int judge(int x,int y,vector<vector<int>>& grid){
        if(x == m) return y;
        if(grid[x][y] == 1){
            if(y+1 == n||grid[x][y+1] == -1) return -1;
            return judge(x+1,y+1,grid);
        }else{
            if(y-1 < 0||grid[x][y-1] == 1) return -1;
            return judge(x+1,y-1,grid);
        }
        return -1;
    }
};
