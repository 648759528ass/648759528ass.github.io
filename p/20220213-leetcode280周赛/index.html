class Solution {
public:
    int countOperations(int num1, int num2) {
        if(num1 == 0 || num2 == 0) return 0;
        if(num1>=num2) return 1 + countOperations(num2,num1-num2);
        return 1 + countOperations(num1,num2-num1);
    }
};

class Solution {
public:
    int minimumOperations(vector<int>& nums) {
        unordered_map<int,int> m1,m2;
        for(int i = 0;i<nums.size();i++){
            if(i%2 == 1){
                ++m1[nums[i]];
            }
            else{
                ++m2[nums[i]];
            }
        }
        priority_queue <pair<int,int>,vector<pair<int,int>>,less<pair<int,int>>>q1,q2;
        for(auto &[k,v] : m1){
            q1.push({v,k});
        }
        for(auto &[k,v] : m2){
            q2.push({v,k});
        }
        if(q1.empty()||q2.empty()) return 0;
        if(q1.top().second == q2.top().second){
            int t1 = q1.top().first;
            int t2 = q2.top().first;
            q1.pop();
            q2.pop();
            t1 += q2.empty()?0:q2.top().first;
            t2 += q1.empty()?0:q1.top().first;
            return nums.size() - max(t1,t2);
        }
        int a = q1.empty()?0:q1.top().first;
        int b = q2.empty()?0:q2.top().first;
        int ans = nums.size() - a - b;
        return  ans ;
    }
};

class Solution {
public:
    long long minimumRemoval(vector<int>& beans) {
        long long ans = LLONG_MAX;
        sort(beans.begin(),beans.end());
        int n = beans.size();
        long long cnt = 0;
        for_each(beans.begin(),beans.end(),[&](int &a){
            cnt+=a;
        });
        for(int i = 0;i<beans.size();i++){
            long long mans = cnt - beans[i]*(long long)(n-i);
            ans = min(ans,mans);
        }
        return ans;
    }
};
