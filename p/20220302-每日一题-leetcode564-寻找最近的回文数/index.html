class Solution {
public:
    long long int find_up(string n){
        auto lln = parse(n);
        int len = n.size();
        string local_tmp = n;
        for(int i = 0;i<len/2;i++){
            local_tmp[len-i-1] = local_tmp[i];  
        }
        auto local_n = parse(local_tmp);
        if(local_n>lln) return local_n;
        if(len%2){
            local_tmp[len/2]+=1;
            if(local_tmp[len/2] == '0'+10){
                for(int i = len/2;i>=0;i--){
                    if(i == 0 && local_tmp[i] == '0'+10){
                        local_tmp[i] = '0';
                        local_tmp = "1"+local_tmp;
                        break;
                    }else if(local_tmp[i] == '0'+10){
                        local_tmp[i-1]+=1;
                        local_tmp[i] = '0';
                    }else{
                        break;
                    }
                }
            }
        }else{
            local_tmp[len/2-1]+=1;
            if(local_tmp[len/2-1] == '0'+10){
                for(int i = len/2 - 1;i>=0;i--){
                    if(i == 0 && local_tmp[i] == '0'+10){
                        local_tmp[i] = '0';
                        local_tmp = "1"+local_tmp;
                        break;
                    }else if(local_tmp[i] == '0'+10){
                        local_tmp[i-1]+=1;
                        local_tmp[i] = '0';
                    }else{
                        break;
                    }
                }
            }
        }
        for(int i = 0;i<local_tmp.size()/2;i++){
            local_tmp[local_tmp.size()-i-1] = local_tmp[i];
        }
        return parse(local_tmp);
    }
    long long int find_low(string n){
        auto lln = parse(n);
        int len = n.size();
        string local_tmp = n;
        for(int i = 0;i<len/2;i++){
            local_tmp[len-i-1] = local_tmp[i];  
        }
        auto local_n = parse(local_tmp);
        if(local_n<lln) return local_n;
        if(len%2){
            local_tmp[len/2]-=1;
            if(local_tmp[len/2] == '0'-1){
                for(int i = len/2;i>=0;i--){
                    if(i == 0 && local_tmp[i] == '0'){
                        if(local_tmp.size() == 1) break;
                        local_tmp = local_tmp.substr(1);
                        break;
                    }else if(local_tmp[i] == '0'-1){
                        local_tmp[i-1]-=1;
                        local_tmp[i] = '9';
                    }else{
                        break;
                    }
                }
            }
        }else{
            local_tmp[len/2-1]-=1;
            if(local_tmp[len/2-1] == '0'-1||(len/2-1 == 0 && local_tmp[len/2-1] == '0')){
                for(int i = len/2 - 1;i>=0;i--){
                    if(i == 0 && local_tmp[i] == '0'){
                        local_tmp[len/2] = '9';
                        local_tmp = local_tmp.substr(1);
                        break;
                    }else if(local_tmp[i] == '0'-1){
                        local_tmp[i-1]-=1;
                        local_tmp[i] = '9';
                    }
                }
            }
        }
        for(int i = 0;i<local_tmp.size()/2;i++){
            local_tmp[local_tmp.size()-i-1] = local_tmp[i];
        }
        return parse(local_tmp);
    }
    long long int parse(string n){
        long long int ret = 0;
        for(int i = 0;i<n.size();i++){
            ret*=10;
            ret += n[i]-'0';
        }
        return ret;
    }
    string nearestPalindromic(string n) {
        auto num = parse(n);
        auto low = find_low(n);
        auto high = find_up(n);
        if(num-low<= high - num){
            return to_string(low);
        }else{
            return to_string(high);
        }
    }
};
