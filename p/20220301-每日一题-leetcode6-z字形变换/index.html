class Solution {
public:
    string convert(string s, int numRows) {
        if(numRows == 1) return s;
        vector<string> ret(numRows,"");
        int lc = 0;
        int add = 1;
        for(int i = 0;i<s.size();i++){
            ret[lc] += s[i];
            lc += add;
            if(lc == numRows){
                add = -1;
                lc-=2;
            }else if(lc == -1){
                add = 1;
                lc+=2;
            }
        }
        string ans = "";
        for(auto & s : ret){
            ans += s;
        }
        return ans;
    }
};
