class Solution {
    int tx[4] = {1,-1,0,0};
    int ty[4] = {0,0,1,-1};
public:
    int numEnclaves(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        queue<pair<int,int>> Q;
        for(int i = 0;i<m;i++){
            if(grid[0][i]) Q.push({0,i}),grid[0][i] = 0;
            if(grid[n-1][i]) Q.push({n-1,i}),grid[n-1][i] = 0;
        }
        for(int i = 0;i<n;i++){
            if(grid[i][0]) Q.push({i,0}),grid[i][0] = 0;
            if(grid[i][m-1]) Q.push({i,m-1}),grid[i][m-1] = 0;
        }
        while(!Q.empty()){
            auto head = Q.front();
            Q.pop();
            for(int i = 0;i<4;i++){
                int dx = head.first + tx[i];
                int dy = head.second + ty[i];
                if(dx>=0&&dx<n&&dy>=0&&dy<m&&grid[dx][dy]){
                    Q.push({dx,dy});
                    grid[dx][dy] = 0;
                }
            }
        }
        int ans = 0;
        for(int i = 0;i<n;i++){
            for(int j = 0;j<m;j++){
                if(grid[i][j]) ans++;
            }
        }
        return ans;
    }
};

class Union{
    public:
        Union(vector<vector<int>>& grid){
            int n = grid.size();
            int m = grid[0].size();
            u.resize(n*m);
            sign.resize(n*m,0);
            for(int i = 0;i<n;i++){
                for(int j = 0;j<m;j++){
                    u[i*m+j] = i*m+j;
                    if(i == 0 || i == n-1 || j == 0||j == m-1){
                        sign[i*m+j] = grid[i][j];
                    }
                }
            }
        };
        int find(int n){
            return n == u[n]?n:u[n] = find(u[n]);
        };
        int check(int n){
            if(u[n] == n) return sign[n];
            if(sign[n]) return 1;
            return sign[n] = check(u[n]);
        };
        void merge(int n,int m){
            int tn = find(n);
            int tm = find(m);
            if(tn != tm){
                u[tn] = tm;
                sign[tm] = sign[tm] | sign[tn] | sign[n] | sign[m];
            }
        }
    private:
        vector<int> u;
        vector<int> sign;
};
class Solution {
    int tx[4] = {1,-1,0,0};
    int ty[4] = {0,0,1,-1};
public:
    int numEnclaves(vector<vector<int>>& grid) {
        Union u(grid);
        int n = grid.size();
        int m = grid[0].size();
        for(int i = 0;i<n;i++){
            for(int j = 0;j<m;j++){
                if(grid[i][j] == 0) continue;
                for(int k = 0;k<4;k++){
                    int dx = i + tx[k];
                    int dy = j + ty[k];
                    if(dx>=0&&dx<n&&dy>=0&&dy<m&&grid[dx][dy])
                    u.merge(i*m+j, dx*m+dy);
                }
            }
        }
        int ans = 0;
        for(int i = 0;i<n;i++){
            for(int j = 0;j<m;j++){
                if(grid[i][j] == 0) continue;
                if(!u.check(i*m+j)) ans++; 
            }
        }
        return ans;
    }
};
