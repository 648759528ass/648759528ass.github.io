class Solution {
public:
    int checkWays(vector<vector<int>>& pairs) {
        unordered_map<int,unordered_set<int>> adj;
        int ans = 0;
        for(auto &v : pairs){
            adj[v[0]].emplace(v[1]);
            adj[v[1]].emplace(v[0]);
        }
        int n = adj.size();
        int root = -1;
        for(auto &[k,v] : adj){
            if(v.size() == n-1){
                root = k;
                break;
            }
        }
        if(root == -1) return 0;
        for(auto &[k,neighbors] : adj){
            if(k == root) continue;
            int curDegree = neighbors.size();
            int parent = -1;
            int parentDegree = INT_MAX;
            for(auto &neighbor : neighbors){
                if(adj[neighbor].size()>=curDegree&&adj[neighbor].size()<=parentDegree){
                    parent = neighbor;
                    parentDegree = adj[neighbor].size();
                }
            }
            if(parent == -1) return 0;
            for(auto &neighbor : neighbors){
                if(neighbor == parent) continue;
                if(adj[parent].find(neighbor) == adj[parent].end()){
                    return 0;
                }
            }
            if(parentDegree == curDegree) ans = max(ans,2);
            else ans = max(ans,1);
        }
        return ans;
    }
};

class Solution {
public:   
// æ€è·¯æœ‰
    int checkWays(vector<vector<int>>& pairs) {
        int maxn = 0;
        for(auto& p : pairs) maxn = max(maxn, max(p[0], p[1]));

        // æ±‚å‡ºå…³ç³»æ•°,åˆå§‹åŒ–å…³ç³»çŸ©é˜µ(conn)
        int deg[maxn + 1], conn[maxn + 1][maxn + 1];
        memset(deg, 0, sizeof(deg));
        memset(conn, 0, sizeof(conn));
        for(auto& p : pairs) {
            conn[p[0]][p[1]] = conn[p[1]][p[0]] = 1;
            deg[p[0]]++, deg[p[1]]++;
        }

        // æ±‚å‡ºå®é™…ç‚¹æ•° n
        int nodes[maxn + 1], n = 0;
        for(int i = 1; i <= maxn; ++i) if(deg[i]) nodes[n++] = i;

        // æŒ‰ç…§ "å…³ç³»æ•°" ä»å¤§åˆ°å°æ’åº
        sort(nodes, nodes + n, [&](int a, int b) {
            return deg[a] > deg[b];
        });

        // æ ¹èŠ‚ç‚¹å¿…é¡»æ˜¯å…¶å®ƒèŠ‚ç‚¹çš„ç¥–å…ˆ
        if(deg[nodes[0]] != n-1) return 0;

        // å¯»æ‰¾çˆ¶èŠ‚ç‚¹ (fa) å’Œç¥–å…ˆèŠ‚ç‚¹ (all_fa)
        int fa[maxn + 1], all_fa[maxn + 1][maxn + 1];
        memset(fa, 0, sizeof(fa));
        memset(all_fa, 0, sizeof(all_fa));
        for(int i = 0; i < n; ++i) {
            for(int j = i-1; j >= 0; --j) {
                if(conn[nodes[i]][nodes[j]]) {
                    fa[nodes[i]] = nodes[j];
                    for(int f = fa[nodes[i]]; f; f = fa[f]) all_fa[nodes[i]][f] = 1;
                    break;
                }
            }
        }

        // æ ¹æ®é¢˜ç›®å®šä¹‰åˆ¤æ–­å³å¯
        int res = 1;
        for(int i = 1; i <= maxn; ++i) {
            for(int j = i + 1; j <= maxn; ++j) {
                if(conn[i][j] && deg[i] == deg[j]) res = 2; // åˆ¤æ–­æ˜¯å¦æœ‰å¤šä¸ªè§£
                if(conn[i][j] != (all_fa[i][j] || all_fa[j][i])) return 0; // åˆ¤æ–­æ˜¯å¦è¿åé¢˜ç›®è¦æ±‚
            }
        }

        return res;
    }
};
