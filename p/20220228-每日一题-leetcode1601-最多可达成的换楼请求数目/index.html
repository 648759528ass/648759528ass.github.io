class Solution {
public:
    int d[20];
    int ans = 0;
    void dfs(int pos,int cnt,int n,vector<vector<int>>& requests){
        if(pos == requests.size()){
            bool ok = true;
            for(int i = 0;i<n;i++){
                if(d[i]!=0) ok = false;
            }
            if(ok) ans = max(ans,cnt);
            return;
        }
        dfs(pos+1, cnt, n , requests);
        d[requests[pos][0]] -=1;
        d[requests[pos][1]] +=1;
        dfs(pos+1, cnt+1, n,requests);
        d[requests[pos][1]] -=1;
        d[requests[pos][0]] +=1;
    }
    int maximumRequests(int n, vector<vector<int>>& requests) {
        memset(d, 0, sizeof(d));
        dfs(0,0,n,requests);
        return ans;
    }
    
};

class Solution {
public:
    int d[20];
    int maximumRequests(int n, vector<vector<int>>& requests) {
        int ans = 0;
        int m = requests.size();
        for(int i = 1;i<1<<m;i++){
            memset(d,0,sizeof(d));
            for(int j = 0;j<m;j++){
                if((i>>j)&1){
                    d[requests[j][0]]--;
                    d[requests[j][1]]++;
                }
            }
            bool ok = 1;
            for(int j = 0;j<n;j++){
                if(d[j]!=0) ok = false;
            }
            if(ok) ans = max(ans,__builtin_popcount(i));
        }
        return ans;
    }
    
};
