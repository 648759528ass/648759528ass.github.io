class Solution {
public:
    vector<int> goodDaysToRobBank(vector<int>& security, int time) {
        vector<int> pre(security.size(),0),r_pre(security.size(),0),ret;
        for(int i = 1;i<security.size();i++){
            if(security[i-1]>=security[i]){
                pre[i] = pre[i-1]+1;
            }
            
        }
        for(int i = security.size()-2;i>=0;i--){
            if(security[i]<=security[i+1]){
                r_pre[i] = r_pre[i+1]+1;
            }
        }
        for(int i = 0;i<security.size();i++){
            if(pre[i]>=time&&r_pre[i]>=time){
                ret.push_back(i);
            }
        }
        return ret;
    }
};

class Solution {
public:
    vector<int> goodDaysToRobBank(vector<int>& security, int time) {
        vector<int> ret;
        int l = 0,r =0,mov = 0;
        while(r<security.size()){
            if(r-l == time*2){
                int idx = (r+l)>>1;
                ret.push_back(idx);
                if(idx+1< security.size() && security[idx] == security[idx+1]){
                    l++;
                }else if(time == 0){
                    l++;
                }else{
                    l = mov;
                }
            }
            if(++r >= security.size()) break;
            if(r<=l+time){
                if(security[r]>security[r-1]){
                    l = r;
                }
            }else if(r == l+time + 1){
                if(security[r]<security[r-1]){
                    l++;
                }
                mov = r;
            }else{
                if(security[r]<security[r-1]){
                    l = mov;
                }else if(security[r]>security[r-1]){
                    mov = r;
                }
            }
            
        }
        return ret;
    }
};
