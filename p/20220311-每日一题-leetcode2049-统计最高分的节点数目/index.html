class Solution {
public:
    int num = 0;
    long long ans = 0;
    int cnt = 0;
    vector<pair<int,int>> V; 
    int countHighestScoreNodes(vector<int>& parents) {
        num = parents.size();
        V.resize(num, {-1,-1});
        for(int i = 1;i<num;i++){
            if(V[parents[i]].first != -1){
                V[parents[i]].second = i;
            }else{
                V[parents[i]].first = i;
            }
        }
        dfs(0);
        return cnt;
    }
    int dfs(int n){
        if(n == -1) return 0;
        int l = dfs(V[n].first);
        int r = dfs(V[n].second);
        long long t = max((long long)(num - l - r - 1),(long long)1)*max((long long)1,(long long)l)*max((long long)1,(long long)r);
        if(t>ans){
            ans = t;
            cnt = 0;
        }
        if(ans == t) cnt++;
        return l + r + 1;
    }
};
