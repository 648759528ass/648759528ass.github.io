class Solution {
public:
    double dp[2][25][25];
    int tx[8] = {-2,-2,-1,-1,1,1,2,2};
    int ty[8] = {-1,1,2,-2,2,-2,-1,1};
    double knightProbability(int n, int k, int row, int column) {
        dp[0][row][column] = 1.0;
        for(int t = 1;t<=k;t++){
            for(int i = 0;i<n;i++){
                for(int j = 0;j<n;j++){
                    double cnt = 0;
                    for(int l = 0;l<8;l++){
                        int dx = i+tx[l];
                        int dy = j+ty[l];
                        if(dx<0||dx>=n||dy<0||dy>=n) continue;
                        cnt += dp[(t-1)&1][dx][dy] * 1.0/8.0;
                    }
                    dp[t&1][i][j] = cnt;
                }
            }
        }
        double cnt = 0;
        for(int i = 0;i<n;i++){
                for(int j = 0;j<n;j++){
                    cnt += dp[k&1][i][j];
                }
            }
            return cnt;
    }
};
