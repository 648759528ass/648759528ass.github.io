class Solution {
public:
    vector<int> platesBetweenCandles(string s, vector<vector<int>>& queries) {
        vector<int> pre(s.size(),0),ret,fx;
        int cnt = 0;
        int i = 0;
        while(i<s.length()&&s[i] != '|') i++;
        for(;i<s.length();i++){
            if(s[i] == '|'){
                fx.push_back(i);
                if(i == 0) continue;
                pre[i] = pre[i-1] + cnt;
                cnt = 0;
            }else{
                pre[i] = pre[i-1];
                cnt++;
            }
        }
        for(auto &q : queries){
            int tidx = lower_bound(fx.begin(), fx.end(), q[0]) - fx.begin();
            if(tidx == fx.size() || fx[tidx] >= q[1]){
                ret.push_back(0);
                continue;
            }
            int lidx = upper_bound(fx.begin(), fx.end(), q[1]) - fx.begin() -1; 
            if(tidx == -1 || fx[lidx] < q[0] || fx[lidx]>q[1]){
                ret.push_back(0);
                continue;
            }
            if(tidx == lidx){
                ret.push_back(0);
            }else{
                ret.push_back(pre[fx[lidx]] - pre[fx[tidx]]);
            }
        }
        return ret;
    }
};

class Solution {
public:
    vector<int> platesBetweenCandles(string s, vector<vector<int>>& queries) {
        vector<int> pre,ret,left(s.size(),-1),right(s.size(),-1);
        int cnt = 0;
        for(int i = 0;i<s.length();i++){
            if(s[i] == '*')
            cnt++;
            pre.push_back(cnt);
        }
        int tag = -1;
        for(int i = 0;i<s.length();i++){
            if(s[i] == '|') tag = i;
            left[i] = tag;
        }
        for(int i = s.size() - 1;i>=0;i--){
            if(s[i] == '|') tag = i;
            right[i] = tag;
        }
        for(auto &q : queries){
            if(right[q[0]] > q[1] || left[q[1]] < q[0]){
                ret.push_back(0);
            }else{
                ret.push_back(pre[left[q[1]]] - pre[right[q[0]]]);
            }
        }
        return ret;
    }
};
