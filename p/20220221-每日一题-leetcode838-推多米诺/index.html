class Solution {
public:
    string pushDominoes(string dominoes) {
        int l = 0;
        for(int i = 0;i<dominoes.size();i++){
            if(dominoes[i] == 'L'){
                if(dominoes[l] == 'R'){
                    for(int j = 0;j<=(i-l)/2;j++){
                        if(l+j == i-j){
                            continue;
                        }
                        dominoes[l+j] = 'R';
                        dominoes[i-j] = 'L';
                    }
                    
                }else{
                    while(l<=i){
                        dominoes[l++] = 'L';
                    }
                }
                l = i+1;
            }else if(dominoes[i] == 'R'){
                if(dominoes[l] == 'R'){
                    while(l<i){
                        dominoes[l++] = 'R';
                    }
                }
                l = i;
            }
        }
        if(dominoes[l] == 'R'){
            while(l<dominoes.size()) dominoes[l++] = 'R';
        }
        return dominoes;
    }
};
